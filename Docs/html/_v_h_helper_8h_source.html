95  static std::array<VkVertexInputAttributeDescription, 4> getAttributeDescriptions() {
96  std::array<VkVertexInputAttributeDescription, 4> attributeDescriptions = {};
97 
98  attributeDescriptions[0].binding = 0;
99  attributeDescriptions[0].location = 0;
100  attributeDescriptions[0].format = VK_FORMAT_R32G32B32_SFLOAT;
101  attributeDescriptions[0].offset = offsetof(vhVertex, pos);
102 
103  attributeDescriptions[1].binding = 0;
104  attributeDescriptions[1].location = 1;
105  attributeDescriptions[1].format = VK_FORMAT_R32G32B32_SFLOAT;
106  attributeDescriptions[1].offset = offsetof(vhVertex, normal);
107 
108  attributeDescriptions[2].binding = 0;
109  attributeDescriptions[2].location = 2;
110  attributeDescriptions[2].format = VK_FORMAT_R32G32B32_SFLOAT;
111  attributeDescriptions[2].offset = offsetof(vhVertex, tangent);
112 
113  attributeDescriptions[3].binding = 0;
114  attributeDescriptions[3].location = 3;
115  attributeDescriptions[3].format = VK_FORMAT_R32G32_SFLOAT;
116  attributeDescriptions[3].offset = offsetof(vhVertex, texCoord);
117 
118  return attributeDescriptions;
119  }
120 
122  bool operator==(const vhVertex& other) const {
123  return pos == other.pos && normal == other.normal && tangent == other.tangent && texCoord == other.texCoord;
124  }
